1. Изменить город всех клиентов из Лондона на Бристоль

UPDATE Customers
SET City = 'Bristol'
WHERE City = 'London'

2. Изменить адрес клиента с CustomerID=18 на новый '15 Tsawassen Blvd'

UPDATE Customers
SET Address = '15 Tsawassen Blvd'
WHERE CustomerID = 18;
